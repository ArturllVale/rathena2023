function	script	Platinum	{
	mes "[Habilidades de Platina]";
	mes "Gostaria de receber as habilidades extras de sua classe?";
	next;
	if (select("^f52a67[+] ^7e2af5Sim, por favor!:^f52a67[+] ^7e2af5Não, Valeu^000000") == 2) {
		mes "[Habilidades de Platina]";
		mes "Tenha um ótimo dia... >.>";
		close;
	}
	callfunc "F_GetPlatinumSkills";
	mes "[Habilidades de Platina]";
	mes "Aí está!!";
	close;
}
